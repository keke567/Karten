cmake_minimum_required(VERSION 3.10)
project(Karten)

set(CMAKE_CXX_STANDARD 17)

# 关键：使用独立版Asio
# 1. 添加Asio头文件路径
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/third_party/asio/asio-1.36.0/include)

# 2. 定义ASIO_STANDALONE宏（重要！）
add_definitions(-DASIO_STANDALONE)

# 3. 创建可执行文件
add_executable(server server.cpp)

# 4. Windows需要链接这些系统库
if(WIN32)
    target_link_libraries(server ws2_32 mswsock)
endif()

# 输出成功信息
message(STATUS "使用Asio独立模式，无需Boost依赖")
