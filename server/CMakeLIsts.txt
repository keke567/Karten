cmake_minimum_required(VERSION 3.10)
project(Karten)

set(CMAKE_CXX_STANDARD 20)

if(MSVC)
    # 对于 Visual Studio，设置源文件使用 UTF-8 编码
    # /utf-8 选项告诉编译器将源文件和执行字符集都视为 UTF-8
    add_compile_options(/utf-8)
    # 设置执行字符集为 UTF-8（可选，/utf-8 已包含此功能）
    add_compile_definitions(_UTF8_SOURCE)
endif()

# 关键：使用独立版Asio
# 1. 添加Asio头文件路径
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/third_party/asio/asio-1.36.0/include)

# 2. 定义ASIO_STANDALONE宏（重要！）
add_definitions(-DASIO_STANDALONE)

# 3. 创建可执行文件
add_executable(server server.cpp session.cpp manager.cpp)

# 4. Windows需要链接这些系统库
if(WIN32)
    target_link_libraries(server ws2_32 mswsock)
endif()

# 输出成功信息
message(STATUS "使用Asio独立模式，无需Boost依赖")
